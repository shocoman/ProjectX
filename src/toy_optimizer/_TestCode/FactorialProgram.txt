fact:
    $t0 = 1    
    $t1 = 1
    getparam $t0
fact_loop:
    $t1 = $t1 * $t0
    $t0 = $t0 - 1
    iftrue $t0 goto fact_loop
    return $t1
main:
    num: .long 0
    res: .long 0
    repeat: .word 0
    just_array: .block 4, word, 'H'

begin:
    putparam "Enter a number:"
    call swrite, 1
    num = call iread, 0

    putparam num
    res = call fact, 1

    putparam res
    call iwrite, 1

    putparam "Repeat? 1 - Yes, 0 - No"
    call swrite, 1

    repeat = call iread, 0
    iftrue repeat goto begin
